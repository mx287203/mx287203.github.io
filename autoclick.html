from pyautogui import *
import keyboard
import threading
import ctypes,sys
s=0
def main():
global s
while 1:
if s:
mouseDown()
sleep(0.01)
mouseUp()
sleep(0.1)
def service():
global s
while 1:
if keyboard.is_pressed('p'):
s=1-s
sleep(0.3)
sleep(0.01)
def is_admin():
try:
return ctypes.windll.shell32.IsUserAnAdmin()
except:
return None
if is_admin():
a = threading.Thread(target=main)
b = threading.Thread(target=service)
a.start()
b.start()
else:
ctypes.windll.shell32.ShellExecuteW(None,u'runas',sys.executable,__file__,None,1)
